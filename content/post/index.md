#removing some symbols
#training data
def clean_text(text):
    sentences = text.lower()
    sentences = re.sub (r'(https|http)?:\/\/(\w|\.|\/|\?|\=|\&|\%)*\b', ' ', sentences, flags=re.MULTILINE)
    sentences = re.sub('[!#?,.:";//|''@-]','',sentences)
    return sentences
    
label_set['text'] = label_set['text'].apply(lambda x: clean_text(x))
